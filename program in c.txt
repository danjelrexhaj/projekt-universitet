#include <stdio.h>

int main() {
    int j, i, n;
    
    printf("Vendos numrin e rreshtave: ");
    scanf("%d", &n);
    
    for (i = 0; i < n; i++) {
        for (j = 0; j<i+1; j ++) {
            printf("*");
        }
        printf("\n");
    }
    
    return 0;
}

#include <stdio.h>

int main() {
    int j, i, n;
    
    printf("Vendos numrin e rreshtave: ");
    scanf("%d", &n);
    
    for (i = 0; i < n; i++) {
        for (j = 0; j<n-i; j ++) {
            printf("*");
        }
        printf("\n");
    }
    
    return 0;

#include <stdio.h>
int main (void){
    int j, i, n;
    printf("Ju lutem vendosni numrin e rreshtave: ");
    scanf("%d",&n);
    for( i = i; i <= n; i ++){
        for( j = 1; j <= n; j ++){
            if( j<=n - i){
                printf(" ");
            }else{
                printf("*");
            }
        }
        printf("\n");
    }
    return 0;
}
